**Timeout cleanup (prevent setState on unmounted component):**
```typescript
useEffect(() => {
  return () => {
    if (saveTimeoutRef.current) clearTimeout(saveTimeoutRef.current);
  };
}, []);
```

**Reusable showSavedFlash helper:**
```typescript
const showSavedFlash = useCallback(() => {
  Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success);
  setSavedVisible(true);
  if (saveTimeoutRef.current) clearTimeout(saveTimeoutRef.current);
  saveTimeoutRef.current = setTimeout(() => setSavedVisible(false), 1500);
}, []);
```

**Save handlers (all on blur):**
```typescript
const handleNotesBlur = useCallback(() => {
  if (notes !== (plant?.notes ?? '')) {
    updatePlant(plant.id, { notes });
    showSavedFlash();
  }
}, [notes, plant, updatePlant, showSavedFlash]);

// Similar pattern for each metadata field:
const handlePurchaseDateBlur = useCallback(() => {
  if (purchaseDate !== (plant?.purchaseDate ?? '')) {
    updatePlant(plant.id, { purchaseDate });
    showSavedFlash();
  }
}, [purchaseDate, plant, updatePlant, showSavedFlash]);
// ... same for purchasePrice, purchaseOrigin, giftFrom
```

**Render structure:**
```
KeyboardAvoidingView (flex:1, behavior=padding/height)
  ScrollView (keyboardShouldPersistTaps="handled")
    View (padding: 16)

      // Notes section
      TextInput (
        multiline, maxLength=1000,
        value=notes, onChangeText=setNotes, onBlur=handleNotesBlur,
        placeholder=t('detail.notes.placeholder'),
        style: minHeight 120, border, rounded
        textAlignVertical="top"
      )

      // Character counter (conditional)
      {notes.length >= COUNTER_THRESHOLD && (
        <Text style={styles.charCounter}>
          {t('detail.notes.charsRemaining', { count: NOTES_MAX - notes.length })}
        </Text>
      )}

      // Saved flash (absolute positioned or below counter, muted green)
      {savedVisible && (
        <Text style={styles.savedConfirm}>{t('detail.notes.saved')}</Text>
      )}

      // Metadata section header
      <Text style={styles.metadataSectionLabel}>
        {t('detail.notes.metadataSection')}
      </Text>

      // 4 metadata fields (inline, always visible, not collapsible)
      MetadataField label=t('detail.notes.purchaseDate') value=purchaseDate onChange=setPurchaseDate onBlur=handlePurchaseDateBlur placeholder="e.g. 2024-03-15"
      MetadataField label=t('detail.notes.purchasePrice') value=purchasePrice onChange=setPurchasePrice onBlur=handlePurchasePriceBlur placeholder="e.g. €12.50"
      MetadataField label=t('detail.notes.purchaseOrigin') value=purchaseOrigin onChange=setPurchaseOrigin onBlur=handlePurchaseOriginBlur placeholder="e.g. IKEA"
      MetadataField label=t('detail.notes.giftFrom') value=giftFrom onChange=setGiftFrom onBlur=handleGiftFromBlur placeholder="e.g. Grandma"
```

**MetadataField** (internal reusable component):
```typescript
function MetadataField({ label, value, onChange, onBlur, placeholder }: {
  label: string;
  value: string;
  onChange: (v: string) => void;
  onBlur: () => void;
  placeholder?: string;
}) {
  return (
    <View style={styles.metadataField}>
      <Text style={styles.metadataLabel}>{label}</Text>
      <TextInput
        style={styles.metadataInput}
        value={value}
        onChangeText={onChange}
        onBlur={onBlur}
        placeholder={placeholder}
        placeholderTextColor="#bbb"
        returnKeyType="done"
      />
    </View>
  );
}
```
