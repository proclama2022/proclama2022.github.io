```typescript
import React, { useState, useEffect, useCallback } from 'react';
import { FlatList, RefreshControl, StyleSheet } from 'react-native';
import { useFocusEffect } from '@react-navigation/native';
import { useRouter } from 'expo-router';
import { useTranslation } from 'react-i18next';

import { ThemedView } from '@/components/Themed';
import { FeedFilterTabs } from '@/components/community/FeedFilterTabs';
import { EmptyFeedState } from '@/components/community/EmptyFeedState';
import { LikesListModal } from '@/components/community/LikesListModal';
import { PostCard } from '@/components/community/PostCard'; // From Phase 13
import { useFeedStore } from '@/stores/feedStore';
import { useAuthStore } from '@/stores/authStore';
import { useColorScheme } from '@/components/useColorScheme';
import Colors from '@/constants/Colors';

export default function CommunityScreen() {
  const { t } = useTranslation();
  const router = useRouter();
  const colorScheme = useColorScheme() ?? 'light';
  const colors = Colors[colorScheme];
  const { user } = useAuthStore();

  const {
    posts,
    filter,
    isLoading,
    isLoadingMore,
    hasMore,
    setFilter,
    fetchPosts,
    fetchMorePosts,
    refreshPosts,
  } = useFeedStore();

  const [likesModalVisible, setLikesModalVisible] = useState(false);
  const [selectedPostId, setSelectedPostId] = useState<string | null>(null);
  const [selectedPostLikeCount, setSelectedPostLikeCount] = useState(0);

  // Fetch posts on mount and when filter changes
  useEffect(() => {
    fetchPosts(user?.id);
  }, [filter]);

  // Refresh on tab focus
  useFocusEffect(
    useCallback(() => {
      refreshPosts(user?.id);
    }, [user?.id])
  );

  const handleFilterChange = (newFilter: typeof filter) => {
    setFilter(newFilter);
  };

  const handleRefresh = useCallback(async () => {
    await refreshPosts(user?.id);
  }, [user?.id, refreshPosts]);

  const handleEndReached = useCallback(() => {
    if (!isLoadingMore && hasMore) {
      fetchMorePosts(user?.id);
    }
  }, [isLoadingMore, hasMore, user?.id, fetchMorePosts]);

  const handleLikeCountPress = (postId: string, likeCount: number) => {
    setSelectedPostId(postId);
    setSelectedPostLikeCount(likeCount);
    setLikesModalVisible(true);
  };

  const handlePostPress = (postId: string) => {
    router.push(`/post/${postId}` as never);
  };

  const handleDiscoverPress = () => {
    // Navigate to search/discovery when available
    // For now, do nothing or show a toast
  };

  const renderPost = useCallback(
    ({ item }) => (
      <PostCard
        post={item}
        onLikeCountPress={() => handleLikeCountPress(item.id, item.like_count)}
        onPostPress={() => handlePostPress(item.id)}
      />
    ),
    []
  );

  if (!isLoading && posts.length === 0) {
    return (
      <ThemedView style={styles.container}>
        <FeedFilterTabs
          activeFilter={filter}
          onFilterChange={handleFilterChange}
        />
        <EmptyFeedState
          filter={filter}
          onDiscoverPress={filter === 'following' ? handleDiscoverPress : undefined}
        />
      </ThemedView>
    );
  }

  return (
    <ThemedView style={styles.container}>
      <FeedFilterTabs
        activeFilter={filter}
        onFilterChange={handleFilterChange}
      />

      <FlatList
        data={posts}
        keyExtractor={(item) => item.id}
        renderItem={renderPost}
        refreshControl={
          <RefreshControl
            refreshing={false}
            onRefresh={handleRefresh}
            tintColor={colors.tint}
          />
        }
        onEndReached={handleEndReached}
        onEndReachedThreshold={0.5}
      />

      <LikesListModal
        visible={likesModalVisible}
        postId={selectedPostId || ''}
        likeCount={selectedPostLikeCount}
        onClose={() => setLikesModalVisible(false)}
      />
    </ThemedView>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
  },
});
```

import React from 'react';
import { View, Image, TouchableOpacity, StyleSheet } from 'react-native';
import { Ionicons } from '@expo/vector-icons';
import { useRouter } from 'expo-router';

import { ThemedText } from '@/components/Themed';
import { Avatar } from '@/components/Avatar';
import { useColorScheme } from '@/components/useColorScheme';
import Colors from '@/constants/Colors';
import type { PostWithAuthor } from '@/lib/supabase/posts';

export interface PostCardProps {
  post: PostWithAuthor;
  onLikeCountPress?: () => void;
  onPostPress?: () => void;
}

export const PostCard: React.FC<PostCardProps> = ({
  post,
  onLikeCountPress,
  onPostPress,
}) => {
  const router = useRouter();
  const colorScheme = useColorScheme() ?? 'light';
  const colors = Colors[colorScheme];

  const handleAuthorPress = () => {
    router.push(`/profile/${post.profiles.id}`);
  };

  const getRelativeTime = (dateString: string): string => {
    const date = new Date(dateString);
    const now = new Date();
    const diffMs = now.getTime() - date.getTime();
    const diffMins = Math.floor(diffMs / 60000);
    const diffHours = Math.floor(diffMs / 3600000);
    const diffDays = Math.floor(diffMs / 86400000);

    if (diffMins < 1) return 'Just now';
    if (diffMins < 60) return `${diffMins}m ago`;
    if (diffHours < 24) return `${diffHours}h ago`;
    if (diffDays < 7) return `${diffDays}d ago`;
    return date.toLocaleDateString();
  };

  return (
    <View style={[styles.card, { backgroundColor: colors.background, borderColor: colors.border }]}>
      {/* Header: Avatar + Name + Time */}
      <TouchableOpacity style={styles.header} onPress={handleAuthorPress}>
        <Avatar uri={post.profiles.avatar_url} size={40} />
        <View style={styles.headerText}>
          <ThemedText style={[styles.authorName, { color: colors.text }]}>
            {post.profiles.display_name}
          </ThemedText>
          <ThemedText style={[styles.timestamp, { color: colors.textMuted }]}>
            {getRelativeTime(post.created_at)}
          </ThemedText>
        </View>
      </TouchableOpacity>

      {/* Photo - Tap for post detail */}
      <TouchableOpacity onPress={onPostPress} activeOpacity={0.9}>
        <Image
          source={{ uri: post.photo_url }}
          style={styles.photo}
          resizeMode="cover"
        />
      </TouchableOpacity>

      {/* Info Section */}
      <View style={styles.info}>
        {post.plant_name && (
          <ThemedText style={[styles.plantName, { color: colors.text }]}>
            {post.plant_name}
          </ThemedText>
        )}
        {post.caption && (
          <ThemedText style={[styles.caption, { color: colors.textSecondary }]} numberOfLines={2}>
            {post.caption}
          </ThemedText>
        )}

        {/* Engagement Row */}
        <View style={styles.engagement}>
          <TouchableOpacity style={styles.engagementItem} onPress={onLikeCountPress}>
            <Ionicons name="heart-outline" size={20} color={colors.textMuted} />
            <ThemedText style={[styles.count, { color: colors.textMuted }]}>
              {post.like_count}
            </ThemedText>
          </TouchableOpacity>
          <TouchableOpacity style={styles.engagementItem} onPress={onPostPress}>
            <Ionicons name="chatbubble-outline" size={20} color={colors.textMuted} />
            <ThemedText style={[styles.count, { color: colors.textMuted }]}>
              {post.comment_count}
            </ThemedText>
          </TouchableOpacity>
        </View>
      </View>
    </View>
  );
};

const styles = StyleSheet.create({
  card: {
    marginHorizontal: 16,
    marginVertical: 8,
    borderRadius: 16,
    borderWidth: 1,
    overflow: 'hidden',
  },
  header: {
    flexDirection: 'row',
    alignItems: 'center',
    padding: 12,
    gap: 12,
  },
  headerText: {
    flex: 1,
  },
  authorName: {

  },

  },

    gap: 6,
  },
  count: {
    fontSize: 14,
  },
});
