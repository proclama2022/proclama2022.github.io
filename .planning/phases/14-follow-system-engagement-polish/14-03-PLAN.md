```typescript
import React, { useState, useEffect, useCallback } from 'react';
import { View, FlatList, Image, TouchableOpacity, StyleSheet, ActivityIndicator } from 'react-native';
import { useRouter } from 'expo-router';
import { useTranslation } from 'react-i18next';
import { Ionicons } from '@expo/vector-icons';

import { ThemedText } from '@/components/Themed';
import { useColorScheme } from '@/components/useColorScheme';
import Colors from '@/constants/Colors';
import { getLikedPosts } from '@/services/likeService';
import type { PostWithAuthor } from '@/lib/supabase/posts';

export interface LikedPostsTabProps {
  userId: string;
}

export const LikedPostsTab: React.FC<LikedPostsTabProps> = ({ userId }) => {
  const { t } = useTranslation();
  const router = useRouter();
  const colorScheme = useColorScheme() ?? 'light';
  const colors = Colors[colorScheme];

  const [posts, setPosts] = useState<PostWithAuthor[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  const [isLoadingMore, setIsLoadingMore] = useState(false);
  const [hasMore, setHasMore] = useState(true);

  useEffect(() => {
    loadPosts();
  }, [userId]);

  const loadPosts = async (cursor?: string) => {
    try {
      if (!cursor) setIsLoading(true);
      else setIsLoadingMore(true);

      const result = await getLikedPosts(userId, 20, cursor);

      if (result.success && result.data) {
        if (cursor) {
          setPosts(prev => [...prev, ...result.data!]);
        } else {
          setPosts(result.data);
        }
        setHasMore(result.data.length === 20);
      }
    } catch (err) {
      console.error('Failed to load liked posts:', err);
    } finally {
      setIsLoading(false);
      setIsLoadingMore(false);
    }
  };

  const handleEndReached = useCallback(() => {
    if (!isLoadingMore && hasMore && posts.length > 0) {
      const lastPost = posts[posts.length - 1];
      loadPosts(lastPost.created_at);
    }
  }, [isLoadingMore, hasMore, posts]);

  const handlePostPress = (postId: string) => {
    router.push(`/post/${postId}` as never);
  };

  if (isLoading) {
    return (
      <View style={styles.centerContainer}>
        <ActivityIndicator size="large" color={colors.tint} />
      </View>
    );
  }

  if (posts.length === 0) {
    return (
      <View style={styles.centerContainer}>
        <Ionicons name="heart-outline" size={48} color={colors.tabIconDefault} />
        <ThemedText style={styles.emptyText}>
          {t('community.noLikedPosts')}
        </ThemedText>
      </View>
    );
  }

  return (
    <FlatList
      data={posts}
      keyExtractor={(item) => item.id}
      numColumns={3}
      renderItem={({ item }) => (
        <TouchableOpacity
          style={styles.gridItem}
          onPress={() => handlePostPress(item.id)}
        >
          <Image
            source={{ uri: item.photo_url }}
            style={styles.thumbnail}
            resizeMode="cover"
          />
        </TouchableOpacity>
      )}
      onEndReached={handleEndReached}
      onEndReachedThreshold={0.5}
      ListFooterComponent={
        isLoadingMore ? (
          <ActivityIndicator size="small" color={colors.tint} />
        ) : null
      }
    />
  );
};

const styles = StyleSheet.create({
  centerContainer: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
    padding: 40,
  },
  emptyText: {
    marginTop: 16,
    opacity: 0.6,
    textAlign: 'center',
  },
  gridItem: {
    flex: 1 / 3,
    aspectRatio: 1,
    padding: 1,
  },
  thumbnail: {
    width: '100%',
    height: '100%',
    borderRadius: 4,
  },
});
```
