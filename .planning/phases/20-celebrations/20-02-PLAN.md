       ```typescript
       useEffect(() => {
         if (!currentToast) {
           setShowCelebration(false);
           setCelebrationType(null);
           return;
         }

         // Define which events deserve confetti
         const celebrationWorthyEvents = ['badge', 'level', 'league_promotion'];
         const shouldCelebrate = celebrationWorthyEvents.includes(currentToast.kind);

         if (shouldCelebrate) {
           // Check cooldown before triggering
           const canCelebrate = useGamificationStore.getState().canTriggerCelebration();
           if (canCelebrate) {
             setCelebrationType(currentToast.kind as CelebrationType);
             setShowCelebration(true);
             useGamificationStore.getState().recordCelebration();
             // Haptic is handled by CelebrationOverlay component
           } else {
             // Cooldown active - skip confetti but still show toast
             setShowCelebration(false);
           }
         } else {
           // For other toasts (except relegation and title), use standard haptic
           if (currentToast.kind !== 'league_relegation' && currentToast.kind !== 'title') {
             Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success).catch(() => undefined);
           }
         }

         const timer = setTimeout(() => {
           dismissToast();
         }, 3000);

         return () => clearTimeout(timer);
       }, [currentToast, dismissToast]);
       ```

    4. **Update JSX:**
       - Replace `LeagueCelebration` with `CelebrationOverlay`
       - Pass `type={celebrationType}` instead of `tier={celebrationTier}`

    5. **Keep LeagueCelebration as backward-compatible wrapper (optional):**
       - Can update LeagueCelebration to simply call CelebrationOverlay with type='league_promotion'
       - Or remove LeagueCelebration import entirely
