1. Create app/(tabs)/camera.tsx:
   ```typescript
   import { CameraView, useCameraPermissions } from 'expo-camera';
   import * as ImagePicker from 'expo-image-picker';
   import { useState } from 'react';
   import { View, TouchableOpacity, Text } from 'react-native';

   export default function CameraScreen() {
     const [permission, requestPermission] = useCameraPermissions();
     const [facing, setFacing] = useState<'back' | 'front'>('back');
     const [capturedUri, setCapturedUri] = useState<string | null>(null);
     const [showOrganSelector, setShowOrganSelector] = useState(false);

     // Permission handling
     if (!permission?.granted) {
       return <PermissionScreen requestPermission={requestPermission} />;
     }

     // Show preview after capture
     if (capturedUri) {
       return (
         <PreviewConfirm
           imageUri={capturedUri}
           onRetake={() => setCapturedUri(null)}
           onConfirm={(organ) => navigateToResults(capturedUri, organ)}
         />
       );
     }

     const takePicture = async () => {
       if (cameraRef.current) {
         const photo = await cameraRef.current.takePictureAsync();
         setCapturedUri(photo.uri);
         setShowOrganSelector(true);
       }
     };

     const pickFromGallery = async () => {
       const result = await ImagePicker.launchImageLibraryAsync({
         mediaTypes: 'images',
         quality: 1,
       });
       if (!result.canceled) {
         setCapturedUri(result.assets[0].uri);
         setShowOrganSelector(true);
       }
     };

     return (
       <View style={{ flex: 1 }}>
         <CameraView style={{ flex: 1 }} facing={facing}>
           {/* Camera overlay with separate buttons */}
           <View style={styles.controls}>
             <TouchableOpacity onPress={pickFromGallery}>
               <Text>From Gallery</Text>
             </TouchableOpacity>
             <TouchableOpacity onPress={takePicture}>
               <Text>Take Photo</Text>
             </TouchableOpacity>
           </View>
         </CameraView>

         <OrganSelector
           visible={showOrganSelector}
           onSelect={(organ) => {
             setShowOrganSelector(false);
             navigateToResults(capturedUri, organ);
           }}
         />
       </View>
     );
   }
   ```

   import { Modal, View, TouchableOpacity, Text } from 'react-native';

   interface Props {
     visible: boolean;
     onSelect: (organ: OrganType) => void;
   }

   const organs: { key: OrganType; label: string }[] = [
     { key: 'leaf', label: 'Leaf' },
     { key: 'flower', label: 'Flower' },
     { key: 'fruit', label: 'Fruit' },
     { key: 'bark', label: 'Bark' },
     { key: 'auto', label: 'Auto-detect' },
   ];

   export default function OrganSelector({ visible, onSelect }: Props) {
     return (
       <Modal visible={visible} animationType="slide">
         <View style={styles.container}>
           <Text style={styles.title}>Select Plant Part</Text>
           {organs.map((organ) => (
             <TouchableOpacity
               key={organ.key}
               onPress={() => onSelect(organ.key)}
             >
               <Text>{organ.label}</Text>
             </TouchableOpacity>
           ))}
         </View>
       </Modal>
     );
   }
