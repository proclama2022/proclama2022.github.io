   ```typescript
   import { usePlantsStore } from '@/stores/plantsStore';
   import { useOnboardingStore } from '@/stores/onboardingStore';
   import { useRouter } from 'expo-router';
   import { View, TouchableOpacity } from 'react-native';
   import Onboarding from '@/components/Onboarding';
   import PlantGrid from '@/components/PlantGrid';
   import { useTranslation } from 'react-i18next';

   export default function HomeScreen() {
     const plants = usePlantsStore(state => state.plants);
     const hasSeenOnboarding = useOnboardingStore(state => state.hasSeenOnboarding);
     const router = useRouter();
     const { t } = useTranslation();

     if (!hasSeenOnboarding) {
       return <Onboarding />;
     }

     if (plants.length === 0) {
       return (
         <View style={styles.emptyContainer}>
           <Text style={styles.emptyTitle}>{t('collection.empty')}</Text>
           <Text style={styles.emptyCTA}>{t('collection.emptyCTA')}</Text>
           <TouchableOpacity
             style={styles.cameraButton}
             onPress={() => router.push('/camera')}
           >
             <Text style={styles.cameraButtonText}>📷 {t('camera.title')}</Text>
           </TouchableOpacity>
         </View>
       );
     }

     return (
       <View style={{ flex: 1 }}>
         <PlantGrid plants={plants} />
         <TouchableOpacity
           style={styles.fab}
           onPress={() => router.push('/camera')}
         >
           <Text style={styles.fabText}>+</Text>
         </TouchableOpacity>
       </View>
     );
   }
   ```

2. Create components/PlantGrid.tsx:
   ```typescript
   import { useState } from 'react';
   import { FlatList, View, TouchableOpacity } from 'react-native';
   import PlantCard from './PlantCard';
   import { SavedPlant } from '@/types';

   interface Props {
     plants: SavedPlant[];
   }

   export default function PlantGrid({ plants }: Props) {
     const [viewMode, setViewMode] = useState<'grid' | 'list'>('grid');

     return (
       <View style={{ flex: 1 }}>
         <View style={styles.header}>
           <TouchableOpacity onPress={() => setViewMode('grid')}>
             <Text style={viewMode === 'grid' ? styles.active : undefined}>
               Grid
             </Text>
           </TouchableOpacity>
           <TouchableOpacity onPress={() => setViewMode('list')}>
             <Text style={viewMode === 'list' ? styles.active : undefined}>
               List
             </Text>
           </TouchableOpacity>
         </View>

         <FlatList
           data={plants}
           key={viewMode}
           numColumns={viewMode === 'grid' ? 2 : 1}
           keyExtractor={(item) => item.id}
           renderItem={({ item }) => (
             <PlantCard plant={item} isGrid={viewMode === 'grid'} />
           )}
           columnWrapperStyle={viewMode === 'grid' ? { gap: 16 } : undefined}
           contentContainerStyle={styles.list}
         />
       </View>
     );
   }
   ```

4. Create app/plant/[id].tsx (Detail screen):
   ```typescript
   import { useLocalSearchParams, useRouter } from 'expo-router';
   import { View, Image, Text, ScrollView, TouchableOpacity } from 'react-native';
   import { usePlantsStore } from '@/stores/plantsStore';
   import { getCareInfo } from '@/services/careDB';

   export default function PlantDetailScreen() {
     const { id } = useLocalSearchParams();
     const router = useRouter();
     const plants = usePlantsStore(state => state.plants);
     const removePlant = usePlantsStore(state => state.removePlant);

     const plant = plants.find(p => p.id === id);
     if (!plant) {
       router.back();
       return null;
     }

     const careInfo = getCareInfo(plant.scientificName);

     const handleDelete = () => {
       removePlant(plant.id);
       router.back();
     };

     return (
       <ScrollView style={{ flex: 1 }}>
         <Image source={{ uri: plant.photo }} style={styles.image} />
         
         <View style={styles.content}>
           <Text style={styles.name}>{plant.commonName || plant.scientificName}</Text>
           {plant.scientificName && (
             <Text style={styles.scientific}>{plant.scientificName}</Text>
           )}

           {plant.nickname && (
             <Text>Nickname: {plant.nickname}</Text>
           )}
           
           {plant.location && (
             <Text>📍 {plant.location}</Text>
           )}

           <Text>Added: {new Date(plant.addedDate).toLocaleDateString()}</Text>

           {careInfo && (
             <View style={styles.care}>
               <Text style={styles.careTitle}>Care Info</Text>
               <Text>💧 Water: every {careInfo.waterFrequencyDays} days</Text>
               <Text>☀️ Sunlight: {careInfo.sunlight}</Text>
               <Text>🌡️ Temperature: {careInfo.tempMin}-{careInfo.tempMax}°C</Text>
               <Text>🌱 Difficulty: {careInfo.difficulty}</Text>
               <Text>🐕 Toxic to pets: {careInfo.toxicPets ? 'Yes' : 'No'}</Text>
               <Text style={styles.tips}>{careInfo.tips.en}</Text>
             </View>
           )}

           <TouchableOpacity onPress={handleDelete} style={styles.deleteButton}>
             <Text style={styles.deleteText}>Delete Plant</Text>
           </TouchableOpacity>
         </View>
       </ScrollView>
     );
   }
   ```
