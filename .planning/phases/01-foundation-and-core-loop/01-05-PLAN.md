   ```typescript
   import { useSettingsStore } from '@/stores/settingsStore';
   import { useTranslation } from 'react-i18next';
   import { changeLanguage } from '@/i18n';

   export default function SettingsScreen() {
     const { language, setLanguage } = useSettingsStore();
     const { t } = useTranslation();

     const handleLanguageChange = async (lang: 'en' | 'it') => {
       setLanguage(lang);
       await changeLanguage(lang);
     };

     return (
       <ThemedView style={styles.container}>
         <ThemedText type="title">{t('settings.title')}</ThemedText>
         
         {/* Language switcher */}
         <View style={styles.section}>
           <ThemedText>{t('settings.language')}</ThemedText>
           <SegmentedControl
             values={['English', 'Italiano']}
             selectedIndex={language === 'en' ? 0 : 1}
             onChange={(event) => {
               const lang = event.nativeEvent.selectedSegmentIndex === 0 ? 'en' : 'it';
               handleLanguageChange(lang);
             }}
           />
         </View>

         {/* PlantNet attribution */}
         <ThemedText style={styles.attribution}>
           {t('settings.attribution')}
         </ThemedText>
       </ThemedView>
     );
   }
   ```
